betterstellaris_wg_subjugation = {
	casus_belli = betterstellaris_cb_subjugation
	set_defender_wargoal = betterstellaris_wg_subjugation
	
	cede_claims = occupied_only
	release_occupied_systems_on_status_quo = no

	allowed_peace_offers = {
		status_quo
		surrender
		demand_surrender
	}

	surrender_acceptance = -100

	potential = {
		always = yes
	}

	on_accept = {
		hidden_effect = {
			from = {
				add_threat = {
					who = root
					amount = 0.5
				}
			}
		}
		ROOT = {
			if = {
				limit = { is_subject = yes }
				FROM = {
					set_subject_of = {
						who = ROOT.overlord
						preset = betterstellaris_preset_subject
					}
				}
			} else = {
				FROM = {
					every_subject = {
						set_subject_of = {
							who = none
						}
					}
					set_subject_of = {
						who = ROOT
						preset = betterstellaris_preset_subject
					}
				}
			}
		}
	}

	ai_weight = {
		weight = 10
	}
}

	}

	on_accept = {
		FROM = {
			set_subject_of {
				who = none
			}
		}
	}

	ai_weight = {
		weight = 100
	}
}
